	for (i = n - 1; i > 0; i--) {

	}

#include <stdio.h>
#define SIZE 10

void main() {
	int i, j, min, temp = 0;
	int n = 10;
	int num[SIZE] = {};

	for (i = 0; i < 10; i++)
	{
		printf("%d번째의 수를 입력하세요.", i + 1);
		scanf_s("%d", &num[i]);
	}

	for (i = 0; i < n-1; i++) {
		min = i;
		for (j = i + 1; j < n; j++) {
			if (num[j] < num[min]) {
				min = j;
			}
			temp = num[min];
			num[min] = num[i];
			num[i] = temp;
		}
	}

	for (i = 0; i < 10; i++)
	{
		printf("%d ", num[i]);
	}
}

#include <stdio.h>

int main() {
    double arr[] = { 98.56, 78.62, 78.69, 89.32, 95.29 };
    double sum = 0;

    for (int i = 0; i < sizeof(arr) / sizeof(double); i++) {
        printf("%.2lf ", arr[i]);
        sum += arr[i];
    }

    printf("\n총합 : %.2lf\n", sum);
    printf("평균 : %.3lf\n", sum / (sizeof(arr) / sizeof(double)));

    return 0;
}
