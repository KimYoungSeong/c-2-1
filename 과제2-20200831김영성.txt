#include <stdio.h>
#define VOLUME(x,y) (3.14*(x)*(x)*(y))

void main() {
	double r, h, bupi;

	printf("반지름과 높이를 입력하세요\n");
	scanf_s("%lf,%lf",&r,&h);

	bupi = VOLUME(r, h);
	printf("부피 : %.2lf", bupi);
}

void main() {
	int won,r1,r2,r3,r4;
	printf("돈을 입력하세요");
	scanf_s("%d", &won);

	r1 = won / 5000;
	r2 = (won - r1*5000) / 1000;
	r3 = (won - (r2*1000 + r1*5000)) / 500;
	r4 = (won - (r3*500 + r2*1000 + r1*5000)) / 100;

	printf("5000원%d개 1000원 %d개 500원 %d개 100원 %d개",r1,r2,r3,r4);
}
