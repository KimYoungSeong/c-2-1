#include <stdio.h>

int triplevalue(int a) {
	a *= 3;

	return a;
}	

void main(){

	int a = 0;

	printf("원하는 값을 하나 입력하세요\n");
	scanf_s("%d", &a);
	
	int res = triplevalue(a);

	printf("그 값의 3배는 %d 입니다.", res);
}

#include <stdio.h>

int maxvalue(int *a,int *b,int *c) {
	
	int max = 0;
	if (*a > *b && *a > *c) 
	{
		max = *a;
	}
	else if (*b > *a && *b > *c)
	{
		max = *b;
	}
	else
	{
		max = *c;
	}
	return max;
}	

void main(){

	int a = 0;
	int b = 0;
	int c = 0;

	printf("원하는 값을 세개 입력하세요\n");
	scanf_s("%d %d %d", &a,&b,&c);
	
	int res = maxvalue(&a,&b,&c);

	printf("제일 큰 수는 %d 입니다.", res);
}
